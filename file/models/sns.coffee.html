SNS creates a URL for service provided by SNS.

module.exports = 
{

  /*
  Facebook
   */
  facebook: {
    createUrl: function(url) {
      return "http://www.facebook.com/share.php?" + (stringify({
        u: url
      }));
    },
    fetchCount: function(url) {
      return $.ajax({
        url: 'https://graph.facebook.com/',
        type: 'get',
        cache: false,
        data: {
          url: url
        },
        dataType: 'jsonp'
      }).then(function(_arg) {
        var d, error, shares;
        error = _arg.error, shares = _arg.shares;
        d = $.Deferred();
        if (error != null) {
          return d.reject(error);
        } else {
          return d.resolve(shares != null ? shares : 0);
        }
      });
    }
  },

  /*
  Twitter
   */
  twitter: {
    createUrl: function(text, url, hashtags) {
      return "http://twitter.com/share?" + (stringify({
        text: text,
        url: url,
        hashtags: hashtags
      }));
    },
    fetchCount: function(url) {
      return $.ajax({
        url: 'http://urls.api.twitter.com/1/urls/count.json',
        type: 'get',
        cache: false,
        data: {
          url: url
        },
        dataType: 'jsonp'
      }).then(function(_arg) {
        var count;
        count = _arg.count;
        return count;
      });
    }
  },

  /*
  Google+
   */
  googlePlus: {
    createUrl: function(url) {
      return "https://plus.google.com/share?" + (stringify({
        url: url
      }));
    },
    fetchCount: function(url) {
      return $.ajax({
        url: "http://query.yahooapis.com/v1/public/yql?env=http://datatables.org/alltables.env&q=" + (encodeURIComponent("SELECT content FROM data.headers WHERE url='https://plusone.google.com/_/+1/fastbutton?hl=ja&url=" + url + "' and ua='" + ua + "'")),
        type: 'get',
        cache: false,
        dataType: 'xml'
      }).then(function(document) {
        var obj, str;
        str = $(document).find('content').text().match(/<script type="text\/javascript">window\.__SSR = ([\s\S]*?);/)[1];
        str = str.replace(/\r?\n/g, '');
        obj = null;
        eval("obj = " + str + ";");
        return obj.ld[1][4];
      });
    }
  },
