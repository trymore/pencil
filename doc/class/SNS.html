  createUrl: function(url) {
    return "http://www.facebook.com/share.php?" + (stringify({
      u: url
    }));
  },
  fetchCount: function(url) {
    return $.ajax({
      url: 'https://graph.facebook.com/',
      type: 'get',
      cache: false,
      data: {
        url: url
      },
      dataType: 'jsonp'
    }).then(function(_arg) {
      var d, error, shares;
      error = _arg.error, shares = _arg.shares;
      d = $.Deferred();
      if (error != null) {
        return d.reject(error);
      } else {
        return d.resolve(shares != null ? shares : 0);
      }
    });
  }
}
